/** * changeSlider() - Core carousel update function * Responsibilities: * 1. Update DOM classes for state management * 2. Reset CSS animations to prevent glitches * 3. Restart auto-play timer * 4. Ensure smooth visual transitions */ const changeSlider = () => { // STEP 1: CLEAN SLATE - Remove all state classes items.forEach(item => { item.classList.remove('active', 'other_1', 'other_2'); }); // STEP 2: APPLY NEW STATE - Add classes based on current indices items[active].classList.add('active'); items[other_1].classList.add('other_1'); items[other_2].classList.add('other_2'); // STEP 3: ANIMATION RESET - Force CSS animations to restart items.forEach(e => { const img = e.querySelector('.image img'); const caption = e.querySelector('.image figcaption'); if (img) { img.style.animation = 'none'; void img.offsetWidth; // Force browser reflow img.style.animation = ''; } // Repeat for caption element if (caption) { caption.style.animation = 'none'; void caption.offsetWidth; caption.style.animation = ''; } }); // STEP 4: AUTO-PLAY RESET - Restart 5-second timer clearInterval(autoPlay); autoPlay = setInterval(() => { next.click(); }, 5000); };

Image Transform Animation - Complete Technical Details

IMAGE SLIDE ANIMATION SYSTEM
CSS
/* * NEXT DIRECTION ANIMATION: Slides images left * Duration: 500ms (0.5 seconds) * Technique: CSS transform (GPU-accelerated) * Variable: --transform-from (set by JavaScript classes) */ @keyframes effectNext { from { transform: translateX(calc(var(--transform-from))); } to { transform: translateX(calc(var(--transform-from) - var(--w-image))); } } /* Animation application for next direction */ .next .item .image img, .next .item .image figcaption { animation: effectNext .5s ease-in-out 1 forwards; } /* Transform starting positions for next animation */ .next .item.active .image { --transform-from: var(--w-image); /* 500px */ } .next .item.other_1 .image { --transform-from: 0px; } .next .item.other_2 .image { --transform-from: calc(var(--w-image) * 2); /* 1000px */ }
